# 代码规范

## 命名规范

```csharp
// 命名空间：功能名称
namespace Inventory
{
    // 类名：PascalCase
    public class InventoryManager
    {
        // 私有字段：_camelCase
        private int _itemCount;
        
        // 公有属性：PascalCase
        public int ItemCount => _itemCount;
        
        // 常量：UPPER_SNAKE_CASE 或 PascalCase
        private const int MAX_SLOTS = 100;
        
        // 方法名：PascalCase
        public void AddItem(ItemData item)
        {
            // 局部变量：camelCase
            var newItem = CreateItem(item);
        }
    }
}
```

## 代码结构模板

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

namespace {功能命名空间}
{
    /// <summary>
    /// {类的简要描述}
    /// </summary>
    public class {类名} : MonoBehaviour
    {
        #region 常量
        private const int MAX_VALUE = 100;
        #endregion
        
        #region 序列化字段
        [SerializeField] private int _value;
        #endregion
        
        #region 私有字段
        private bool _isInitialized;
        #endregion
        
        #region 公有属性
        public int Value => _value;
        #endregion
        
        #region 事件
        public event Action<int> OnValueChanged;
        #endregion
        
        #region Unity 生命周期
        private void Awake() { Initialize(); }
        private void OnDestroy() { Cleanup(); }
        #endregion
        
        #region 公有方法
        /// <summary>
        /// {方法描述}
        /// </summary>
        public void DoSomething() { }
        #endregion
        
        #region 私有方法
        private void Initialize() { _isInitialized = true; }
        private void Cleanup() { }
        #endregion
    }
}
```
